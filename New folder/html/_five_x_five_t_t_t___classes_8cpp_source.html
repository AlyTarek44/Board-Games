240int FxFTTT_board::minimax(int depth, bool is_max, int alpha, int beta) {
241 auto moves = get_available_moves();
242
243 // Stop if no moves left or depth limit reached
244 if (depth == 0 || moves.empty()) {
245 return evaluate_board();
246 }
247

249 int best = -100000;
