31bool FourByFour_Board::update_board(Move<char> *move) {
32 // We know in this game we always pass FourByFour_Move
33 FourByFour_Move *m = static_cast<FourByFour_Move *>(move);
34
35 int fr = m->get_from_r();
36 int fc = m->get_from_c();
37 int tr = m->get_to_r();
38 int tc = m->get_to_c();
39 char symbol = m->get_symbol();
40
41 // 1) Bounds check
42 if (fr < 0 || fr >= 4 || fc < 0 || fc >= 4) {
43 cout << "Invalid move: source cell is outside the board.\n";
44 return false;
45 }
46 if (tr < 0 || tr >= 4 || tc < 0 || tc >= 4) {
47 cout << "Invalid move: destination cell is outside the board.\n";
48 return false;
49 }
50
51 // 2) Source must contain player's symbol
52 if (board[fr][fc] != symbol) {
53 if (board[fr][fc] == '.') {
54 cout << "Invalid move: source cell is empty.\n";
55 } else {
56 cout << "Invalid move: source cell does not contain your token.\n";
57 }
58 return false;
59 }
60
61 // 3) Destination must be empty
62 if (board[tr][tc] != '.') {
63 cout << "Invalid move: destination cell is not empty.\n";
64 return false;
65 }
66
67 // 4) Movement rule: one step horizontally or vertically
68 int dr = abs(fr - tr);
69 int dc = abs(fc - tc);
70
71 if (dr == 0 && dc == 0) {
72 cout << "Invalid move: you must move to a different cell.\n";
73 return false;
74 }
75
76 bool one_step_orthogonal =
77 (dr == 1 && dc == 0) ||
78 (dr == 0 && dc == 1);
79
80 if (!one_step_orthogonal) {
81 cout << "Invalid move: you can only move one step up, down, left, or right.\n";
82 return false;
83 }
84
85 // 5) Apply the move
86 board[tr][tc] = board[fr][fc];
87 board[fr][fc] = '.';
88
89 n_moves++;
90 return true;
91}

93bool FourByFour_Board::check_win_for_symbol(char s) {
94 // Horizontal 3 in a row
95 for (int r = 0; r < 4; ++r) {
96 for (int c = 0; c <= 4 - 3; ++c) {
97 if (board[r][c] == s &&
98 board[r][c + 1] == s &&
99 board[r][c + 2] == s)
100 return true;
101 }
102 }
103
104 // Vertical 3 in a column
105 for (int c = 0; c < 4; ++c) {
106 for (int r = 0; r <= 4 - 3; ++r) {
107 if (board[r][c] == s &&
108 board[r + 1][c] == s &&
109 board[r + 2][c] == s)
110 return true;
111 }
112 }
113
114 // Main diagonal direction (3-length)
115 if (board[0][0] == s && board[1][1] == s && board[2][2] == s) return true;
116 if (board[1][1] == s && board[2][2] == s && board[3][3] == s) return true;
117 if (board[0][1] == s && board[1][2] == s && board[2][3] == s) return true;
118 if (board[1][0] == s && board[2][1] == s && board[3][2] == s) return true;
119
120 // Anti-diagonal direction
121 if (board[0][3] == s && board[1][2] == s && board[2][1] == s) return true;
122 if (board[1][2] == s && board[2][1] == s && board[3][0] == s) return true;
123 if (board[0][2] == s && board[1][1] == s && board[2][0] == s) return true;
124 if (board[1][3] == s && board[2][2] == s && board[3][1] == s) return true;
125
126 return false;
127}

146vector<FourByFour_Move> FourByFour_Board::get_valid_moves(char symbol) {
147 vector<FourByFour_Move> moves;
148
149 // Directions: up, down, left, right
150 int dr[4] = {-1, 1, 0, 0};
151 int dc[4] = {0, 0, -1, 1};
152
153 for (int r = 0; r < 4; ++r) {
154 for (int c = 0; c < 4; ++c) {
155 if (board[r][c] != symbol) continue;
156
157 for (int k = 0; k < 4; ++k) {
158 int nr = r + dr[k];
159 int nc = c + dc[k];
160
161 // Check bounds
162 if (nr < 0 || nr >= 4 || nc < 0 || nc >= 4) continue;
163
164 // Must be empty destination
165 if (board[nr][nc] == '.') {
166 moves.emplace_back(r, c, nr, nc, symbol);
167 }
168 }
169 }
170 }
171
172 return moves;
173}

206 return -1000 - depth; // later loss is better

209 return 0;

213
214 if (moves.empty())
215 return 0;
216
217 if (isMaximizing) {

219

222 int fc = move.get_from_c();

224 int tc = move.get_to_c();
225
