32bool Memory_Board::update_board(Move<char>* move) {
33 int x = move->get_x();
34 int y = move->get_y();
35 char mark = move->get_symbol();
36
37 // Validate against boundaries and the REAL board state
38 if (x < 0 || x >= 3 || y < 0 || y >= 3 || real_board[x][y] != blank_symbol) {
39 return false;
40 }
41
42 n_moves++;
43 real_board[x][y] = toupper(mark);
44 return true;
45}

47vector<vector<char>> Memory_Board::get_board_matrix() const {
48 // Generate the "Fog of War" view
49 vector<vector<char>> hidden_board = vector<vector<char>>(3, vector<char>(3, blank_symbol));
50
51 for (int i = 0; i < 3; ++i) {
52 for (int j = 0; j < 3; ++j) {
53 if (real_board[i][j] != blank_symbol) {
54 hidden_board[i][j] = hidden_symbol; // Mask occupied cells
55 }
56 }
57 }
58 return hidden_board;
59}

61vector<vector<char>> Memory_Board::get_move_display_board(int move_x, int move_y, char symbol) const {
62 // Generate view revealing ONLY the current move
63 vector<vector<char>> move_board = vector<vector<char>>(3, vector<char>(3, blank_symbol));
64
65 for (int i = 0; i < 3; ++i) {
66 for (int j = 0; j < 3; ++j) {
67 if (real_board[i][j] != blank_symbol) {
68 if (i == move_x && j == move_y) {
69 move_board[i][j] = symbol; // Reveal just played
70 } else {
71 move_board[i][j] = hidden_symbol; // Keep others hidden
72 }
73 }
74 }
75 }
76 return move_board;
77}

79bool Memory_Board::is_win(Player<char>* player) {
80 const char sym = player->get_symbol();
81 auto all_equal = [&](char a, char b, char c) {
82 return a == b && b == c && a != blank_symbol;
83 };
84
85 // Check on the REAL board
86 for (int i = 0; i < 3; ++i) {
87 if ((all_equal(real_board[i][0], real_board[i][1], real_board[i][2]) && real_board[i][0] == sym) ||
88 (all_equal(real_board[0][i], real_board[1][i], real_board[2][i]) && real_board[0][i] == sym))
89 return true;
90 }
91 if ((all_equal(real_board[0][0], real_board[1][1], real_board[2][2]) && real_board[1][1] == sym) ||
92 (all_equal(real_board[0][2], real_board[1][1], real_board[2][0]) && real_board[1][1] == sym))
93 return true;
94
95 return false;
96}

143void Memory_UI::run_game(Player<char>* players[2]) {
144 display_board_matrix(memory_board->get_board_matrix());
145 Player<char>* currentPlayer = players[0];
146
147 while (true) {
148 for (int i : {0, 1}) {
149 currentPlayer = players[i];
150
151 Move<char>* move = get_move(currentPlayer);
152
153 // Keep asking until valid move
154 while (!memory_board->update_board(move)) {
155 cout << "Invalid move! Try again.\n";
156 delete move;
157 move = get_move(currentPlayer);
158 }
159
160 // Briefly show what happened
161 cout << "\nYour move was placed at (" << move->get_x() << ", " << move->get_y() << "):\n";
162 display_board_matrix(memory_board->get_move_display_board(move->get_x(), move->get_y(), currentPlayer->get_symbol()));
163
164 cout << "Remember your move...\n\n";
165 this_thread::sleep_for(chrono::milliseconds(1100)); // Wait 1.1 seconds
166
167 clear_screen(); // Wipe memory from screen
168
169 delete move;
170
171 if (memory_board->is_win(currentPlayer)) {
172 display_message(currentPlayer->get_name() + " wins!");
173 return;
174 }
175 if (memory_board->is_draw(currentPlayer)) {
176 display_message("Draw!");
177 return;
178 }
179 }
180 }
