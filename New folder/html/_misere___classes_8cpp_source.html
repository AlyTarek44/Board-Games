46bool Misere_Board::player_has_three(char symbol) const {
47 // Check rows
48 for (int i = 0; i < 3; ++i)
49 if (board[i][0] == symbol && board[i][1] == symbol && board[i][2] == symbol) return true;
50 // Check cols
51 for (int j = 0; j < 3; ++j)
52 if (board[0][j] == symbol && board[1][j] == symbol && board[2][j] == symbol) return true;
53 // Check diagonals
54 if (board[0][0] == symbol && board[1][1] == symbol && board[2][2] == symbol) return true;
55 if (board[0][2] == symbol && board[1][1] == symbol && board[2][0] == symbol) return true;
56 return false;
57}

112bool Misere_Board::placement_creates_three(int r, int c, char symbol) const {
113 // Temporarily checks if a move at (r,c) completes a line for 'symbol'
114 // Check row r
115 bool row_three = true;
116 for (int j = 0; j < 3; ++j) {
117 char val = (j == c) ? symbol : board[r][j];
118 if (val != symbol) { row_three = false; break; }
119 }
120 if (row_three) return true;
121
122 // Check column c
123 bool col_three = true;
124 for (int i = 0; i < 3; ++i) {
125 char val = (i == r) ? symbol : board[i][c];
126 if (val != symbol) { col_three = false; break; }
127 }
128 if (col_three) return true;
129
130 // Diagonals
131 if (r == c) {
132 bool diag = true;
133 for (int i = 0; i < 3; ++i) {
134 char val = (i == r) ? symbol : board[i][i];
135 if (val != symbol) { diag = false; break; }
136 }
137 if (diag) return true;
138 }
139 if (r + c == 2) {
140 bool diag = true;
141 for (int i = 0; i < 3; ++i) {
142 int j = 2 - i;
143 char val = (i == r && j == c) ? symbol : board[i][j];
144 if (val != symbol) { diag = false; break; }
145 }
146 if (diag) return true;
147 }
148 return false;
149}

158Move<char>* Misere_AIPlayer::get_smart_move(Misere_Board* board, char opponent_symbol) {
159 cout << "AI " << get_name() << " is thinking (full backtracking)...\n";
160
161 vector<pair<int,int>> empties = board->get_empty_cells();
162 if (empties.empty()) return new Move<char>(0,0,get_symbol());
163
164 int best_score = -2;
165 vector<pair<int,int>> best_moves;
166
167 for (auto mv : empties) {
168 int r = mv.first, c = mv.second;
169
170 // 1. Immediate Suicidal Check
171 // If playing here makes ME create 3-in-a-row, I lose instantly. Score = -1 (Bad).
172 board->make_temp_move(r, c, get_symbol());
173 if (board->placement_creates_three(r, c, get_symbol())) {
174 board->undo_temp_move(r, c);
175 if (-1 > best_score) {
176 best_score = -1;
177 best_moves.clear();
178 best_moves.push_back(mv);
179 } else if (-1 == best_score) {
180 best_moves.push_back(mv);
181 }
182 continue;
183 }
184
185 // 2. Otherwise, see if this leads to a win down the line
186 int score = backtrack_minimax(board, opponent_symbol);
187 board->undo_temp_move(r, c);
188
189 if (score > best_score) {
190 best_score = score;
191 best_moves.clear();
192 best_moves.push_back(mv);
193 } else if (score == best_score) {
194 best_moves.push_back(mv);
195 }
196 }
197
198 pair<int,int> chosen = tiebreak_choose(best_moves);
199 cout << "AI " << get_name() << " selected (" << chosen.first << "," << chosen.second << ") with score " << best_score << "\n";
200 return new Move<char>(chosen.first, chosen.second, get_symbol());
201}

203int Misere_AIPlayer::backtrack_minimax(Misere_Board* board, char current_symbol) {
204 auto empties = board->get_empty_cells();
205 if (empties.empty()) return 0; // Draw
206
207 bool is_ai_turn = (current_symbol == this->get_symbol());
208 int best = is_ai_turn ? -2 : 2;
209
210 for (auto mv : empties) {
211 int r = mv.first, c = mv.second;
212 board->make_temp_move(r, c, current_symbol);
213
214 // Does this move force the current player to lose?
215 if (board->placement_creates_three(r, c, current_symbol)) {
216 // Mover loses.
217 // If AI moved and lost -> Outcome -1
218 // If Opponent moved and lost -> Outcome +1
219 int outcome = (current_symbol == this->get_symbol()) ? -1 : 1;
220
221 board->undo_temp_move(r, c);
222
223 if (is_ai_turn) best = max(best, outcome);
224 else best = min(best, outcome);
225
226 // Pruning
227 if (is_ai_turn && best == 1) return 1;
228 if (!is_ai_turn && best == -1) return -1;
229 continue;
230 }
231
232 // Recurse
233 char next = (current_symbol == 'X') ? 'O' : 'X';
234 int score = backtrack_minimax(board, next);
235 board->undo_temp_move(r, c);
236
237 if (is_ai_turn) best = max(best, score);
238 else best = min(best, score);
239
240 if (is_ai_turn && best == 1) return 1;
241 if (!is_ai_turn && best == -1) return -1;
242 }
243
244 if (best == -2 || best == 2) return 0;
245 return best;
246}

248pair<int,int> Misere_AIPlayer::tiebreak_choose(const vector<pair<int,int>>& choices) {
249 pair<int,int> center = {1,1};
250 for (auto p : choices) if (p == center) return p; // Prefer Center
251
252 vector<pair<int,int>> corners = { {0,0}, {0,2}, {2,0}, {2,2} };
253 for (auto c : corners) {
254 for (auto p : choices) if (p == c) return p; // Prefer Corners
255 }
256 return choices.front(); // Then Sides
257}
