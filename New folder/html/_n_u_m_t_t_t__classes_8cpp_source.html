33bool NumTTT_Board::update_board(Move<int>* move) {
34 int x = move->get_x();
35 int y = move->get_y();
36 int number = move->get_symbol(); // The number played
37
38 // 1. Check Bounds
39 if (x < 0 || x >= rows || y < 0 || y >= columns) {
40 if (x != -1) cout << "Invalid move: Position out of bounds.\n";
41 return false;
42 }
43
44 // 2. Check Empty
45 if (board[x][y] != blank_symbol) {
46 cout << "Invalid move: Cell occupied.\n";
47 return false;
48 }
49
50 // 3. Validate Number Ownership
51 bool is_odd_turn = (n_moves % 2 == 0); // Player 1 (Odd) always starts
52
53 if (is_odd_turn) {
54 auto it = find(odd_nums.begin(), odd_nums.end(), number);
55 if (it != odd_nums.end()) {
56 odd_nums.erase(it); // Consume number
57 } else {
58 cout << "Invalid move: " << number << " is not available/valid for Odd player.\n";
59 return false;
60 }
61 } else {
62 auto it = find(even_nums.begin(), even_nums.end(), number);
63 if (it != even_nums.end()) {
64 even_nums.erase(it); // Consume number
65 } else {
66 cout << "Invalid move: " << number << " is not available/valid for Even player.\n";
67 return false;
68 }
69 }
70
71 // 4. Apply
72 board[x][y] = number;
73 n_moves++;
74 return true;
75}

77bool NumTTT_Board::check_all_lines() {
78 int sum;
79 // Rows
80 for (int i = 0; i < 3; ++i) {
81 if (board[i][0] && board[i][1] && board[i][2]) {
82 sum = board[i][0] + board[i][1] + board[i][2];
83 if (sum == 15) return true;
84 }
85 }
86 // Columns
87 for (int j = 0; j < 3; ++j) {
88 if (board[0][j] && board[1][j] && board[2][j]) {
89 sum = board[0][j] + board[1][j] + board[2][j];
90 if (sum == 15) return true;
91 }
92 }
93 // Diagonals
94 if (board[0][0] && board[1][1] && board[2][2]) {
95 sum = board[0][0] + board[1][1] + board[2][2];
96 if (sum == 15) return true;
97 }
98 if (board[0][2] && board[1][1] && board[2][0]) {
99 sum = board[0][2] + board[1][1] + board[2][0];
100 if (sum == 15) return true;
101 }
102
103 return false;
104}
