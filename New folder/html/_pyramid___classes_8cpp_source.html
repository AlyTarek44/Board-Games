21bool Pyramid_Board::is_valid_position(int row, int col) {
22 // Row 0: only col 2 (Top)
23 if (row == 0 && col == 2) return true;
24
25 // Row 1: cols 1,2,3 (Middle)
26 if (row == 1 && col >= 1 && col <= 3) return true;
27
28 // Row 2: cols 0,1,2,3,4 (Base)
29 if (row == 2 && col >= 0 && col <= 4) return true;
30
31 return false;
32}

70bool Pyramid_Board::check_win_symbol(char s) {
71 // Horizontal lines
72 if (board[1][1] == s && board[1][2] == s && board[1][3] == s) return true;
73 if (board[2][0] == s && board[2][1] == s && board[2][2] == s) return true;
74 if (board[2][1] == s && board[2][2] == s && board[2][3] == s) return true;
75 if (board[2][2] == s && board[2][3] == s && board[2][4] == s) return true;
76
77 // Vertical line (center column)
78 if (board[0][2] == s && board[1][2] == s && board[2][2] == s) return true;
79
80 // Diagonals
81 if (board[0][2] == s && board[1][1] == s && board[2][0] == s) return true; // /
82 if (board[0][2] == s && board[1][3] == s && board[2][4] == s) return true; // \
83
84 return false;
85}

114int Pyramid_Board::minimax(int depth, bool isMaximizing, char aiSymbol, char humanSymbol) {
115 if (check_win_symbol(aiSymbol)) return 10 - depth;
116 if (check_win_symbol(humanSymbol)) return depth - 10;
117
118 vector<pair<int, int> > validMoves = get_valid_moves();
119 if (validMoves.empty()) return 0; // Draw
120
121 if (isMaximizing) {
122 int bestScore = -1000;
123 for (auto &move: validMoves) {
124 board[move.first][move.second] = aiSymbol;
125 int score = minimax(depth + 1, false, aiSymbol, humanSymbol);
126 board[move.first][move.second] = '.'; // Backtrack
127 bestScore = max(bestScore, score);
128 }
129 return bestScore;
130 } else {
131 int bestScore = 1000;
132 for (auto &move: validMoves) {
133 board[move.first][move.second] = humanSymbol;
134 int score = minimax(depth + 1, true, aiSymbol, humanSymbol);
135 board[move.first][move.second] = '.'; // Backtrack
136 bestScore = min(bestScore, score);
137 }
138 return bestScore;
139 }

142pair<int, int> Pyramid_Board::get_best_move(char aiSymbol) {
143 char humanSymbol = (aiSymbol == 'X') ? 'O' : 'X';
144 vector<pair<int, int> > validMoves = get_valid_moves();
145
146 int bestScore = -1000;
147 pair<int, int> bestMove = {-1, -1};
148
149 for (auto &move: validMoves) {
150 board[move.first][move.second] = aiSymbol;
151 int score = minimax(0, false, aiSymbol, humanSymbol);
152 board[move.first][move.second] = '.';
153
154 if (score > bestScore) {
155 bestScore = score;
156 bestMove = move;
157 }
158 }
159 return bestMove;
160}
