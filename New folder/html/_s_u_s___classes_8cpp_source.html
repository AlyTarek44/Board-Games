46int SUS_Board::check_and_score_sequences(int last_x, int last_y, char symbol) {
47 int points_earned = 0;
48 auto all_lines = get_all_lines();
49
50 for (auto& line : all_lines) {
51 int x1 = line[0].first, y1 = line[0].second;
52 int x2 = line[1].first, y2 = line[1].second;
53 int x3 = line[2].first, y3 = line[2].second;
54
55 // Optimization: Only check lines that include the last move
56 bool contains_last_move = (x1 == last_x && y1 == last_y) ||
57 (x2 == last_x && y2 == last_y) ||
58 (x3 == last_x && y3 == last_y);
59
60 if (contains_last_move) {
61 char a = board[x1][y1];
62 char b = board[x2][y2];
63 char c = board[x3][y3];
64
65 if (a != blank_symbol && b != blank_symbol && c != blank_symbol) {
66 if (a == 'S' && b == 'U' && c == 'S') {
67 points_earned++;
68 }
69 }
70 }
71 }
72 return points_earned;
73}

136int SUS_Board::evaluate_move(int x, int y, char symbol, int depth) {
137 char original_value = board[x][y];
138 board[x][y] = symbol;
139
140 int immediate_ai_score = check_and_score_sequences(x, y, symbol);
141
142 // Simple lookahead to see if opponent can score next
143 char opponent_symbol = (symbol == player1_letter) ? player2_letter : player1_letter;
144 int opponent_scoring_risk = 0;
145
146 if (depth > 0) {
147 auto valid_moves = get_valid_moves();
148 for (auto& move : valid_moves) {
149 int opp_x = move.first;
150 int opp_y = move.second;
151 char opp_original = board[opp_x][opp_y];
152 board[opp_x][opp_y] = opponent_symbol;
153
154 int score = check_and_score_sequences(opp_x, opp_y, opponent_symbol);
155 board[opp_x][opp_y] = opp_original;
156
157 if (score > 0) {
158 opponent_scoring_risk = score;
159 break;
160 }
161 }
162 }
163
164 board[x][y] = original_value;
165
166 if (immediate_ai_score > 0) return 1000; // Priority: Score points
167 else if (opponent_scoring_risk > 0) return 100; // Block opponent
168 else return 500; // Neutral
169}
