28bool Ultimate_Board::update_board(Move<char>* move) {
29 int main_idx = move->get_x();
30 int sub_idx = move->get_y();
31 char sym = move->get_symbol();
32
33 if (main_idx < 0 || main_idx >= 9 || sub_idx < 0 || sub_idx >= 9) return false;
34 // Cannot play in a claimed or full sub-board
35 if (main_claims[main_idx] != ' ') return false;
36
37 int gr, gc; unpack_indices(main_idx, sub_idx, gr, gc);
38 if (board[gr][gc] != blank_symbol) return false;
39
40 // Check constraints
41 if (forced_main != -1 && forced_main != main_idx) return false;
42
43 board[gr][gc] = sym;
44 ++n_moves;
45
46 // Check if this move won the sub-board
47 char winner = check_subboard_winner(main_idx);
48 if (winner != ' ' && main_claims[main_idx] == ' ') main_claims[main_idx] = winner;
49 else if (subboard_full(main_idx) && main_claims[main_idx] == ' ') main_claims[main_idx] = 'D';
50
51 // Update forced_main for next player.
52 // Rule: Next player must play in the sub-board corresponding to the cell index played in the current sub-board.
53 // However, if that target sub-board is full/claimed, they can play anywhere.
54 // Note: The code provided sets forced_main = -1 (removed forcing rule comment).
55 // If you want standard rules, uncomment logic here. Current logic allows free movement.
56 forced_main = -1;
57
58 return true;
59}

96char Ultimate_Board::check_subboard_winner(int main_idx) const {
97 int mr = main_idx / 3, mc = main_idx % 3;
98 auto get = [&](int sr, int sc)->char {
99 int gr = mr*3 + sr, gc = mc*3 + sc;
100 return board[gr][gc];
101 };
102 // Standard TTT check
103 for (int r = 0; r < 3; ++r) {
104 char a = get(r,0), b = get(r,1), c = get(r,2);
105 if (a != blank_symbol && a == b && b == c) return a;
106 }
107 for (int c = 0; c < 3; ++c) {
108 char a = get(0,c), b = get(1,c), c2 = get(2,c);
109 if (a != blank_symbol && a == b && b == c2) return a;
110 }
111 if (get(0,0) != blank_symbol && get(0,0) == get(1,1) && get(1,1) == get(2,2)) return get(0,0);
112 if (get(0,2) != blank_symbol && get(0,2) == get(1,1) && get(1,1) == get(2,0)) return get(0,2);
113 return ' ';
114}

124char Ultimate_Board::check_three_in_row_on_main(char who) const {

130 if (main_claims[0] == who && main_claims[4] == who && main_claims[8] == who) return who;
131 if (main_claims[2] == who && main_claims[4] == who && main_claims[6] == who) return who;

168Move<char>* Ultimate_AIPlayer::get_simple_move(Ultimate_Board* board) {
