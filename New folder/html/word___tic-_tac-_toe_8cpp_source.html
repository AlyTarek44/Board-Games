114Move<char>* word_xo_board::find_best_move() {
115 vector<Move<char>*> safe_moves;
116
117 for (int i = 0; i < rows; ++i) {
118 for (int j = 0; j < columns; ++j) {
119 if (board[i][j] == blank_symbol) {
120 // Try every letter 'A'-'Z'
121 for (char letter = 'A'; letter <= 'Z'; ++letter) {
122
123 // 1. Can AI win immediately?
124 simulate_move(i, j, letter);
125 if (is_valid_word()) {
126 undo_move(i, j, letter);
127 return new Move<char>(i, j, letter);
128 }
129
130 // 2. Does this move let Human win next turn?
131 bool human_can_win = false;
132 for (int hi = 0; hi < rows; ++hi) {
133 for (int hj = 0; hj < columns; ++hj) {
134 if (board[hi][hj] == blank_symbol) {
135 for (char hletter = 'A'; hletter <= 'Z'; ++hletter) {
136 simulate_move(hi, hj, hletter);
137 if (is_valid_word()) human_can_win = true;
138 undo_move(hi, hj, hletter);
139 }
140 }
141 }
142 }
143 undo_move(i, j, letter);
144
145 if (!human_can_win) {
146 safe_moves.push_back(new Move<char>(i, j, letter));
147 }
148 }
149 }
150 }
151 }
152
153 if (!safe_moves.empty()) return safe_moves[0];
154
155 // Fallback
156 for (int i = 0; i < rows; ++i)
157 for (int j = 0; j < columns; ++j)
158 if (board[i][j] == blank_symbol)
159 return new Move<char>(i, j, 'A');
160
161 return nullptr;
162}
